# Source files
add_library(boost-sml-utils INTERFACE)

# Include directories
target_include_directories(boost-sml-utils 
    INTERFACE 
        $<INSTALL_INTERFACE:include>
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
)

# Link dependancies
target_link_libraries(boost-sml-utils
    INTERFACE
        sml
        estd-traits
        boost-headers
)
